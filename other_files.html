char *strtok(char *str, const char *delim) {
    static char *p;
    if (str) p = str;
    else str = p;
    str += strspn(str, delim);
    if (!*str) return NULL;
    p = str + strcspn(str, delim);
    if (*p) *p++ = 0;
    return str;
}

void bubble_sort(int arr[], int n) {
    int i, j, temp;
    for (i = 0; i < n-1; i++) {
        for (j = 0; j < n-i-1; j++) {
            if (arr[j] > arr[j+1]) {
                temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
}
